# =============================================
# 1. ПРОВЕРКА И НАСТРОЙКА ОС
# =============================================

# Имя компьютера

# Часовой пояс
timedatectl

# Язык системы
locale

# =============================================
# 2. ПРОВЕРКА ДРАЙВЕРОВ (Guest Additions)
# =============================================

lsmod | grep vbox

# =============================================
# 3. НАСТРОЙКА КЛАВИАТУРЫ (русская + английская)
# =============================================

sudo sed -i 's/XKBLAYOUT=.*/XKBLAYOUT="us,ru"/' /etc/default/keyboard
sudo sed -i 's/XKBOPTIONS=.*/XKBOPTIONS="grp:alt_shift_toggle"/' /etc/default/keyboard
sudo dpkg-reconfigure -f noninteractive keyboard-configuration
cat /etc/default/keyboard   # проверка



# =============================================
# 4. НАСТРОЙКА SSH (установка + включение, если надо)
# =============================================
# Проверяем, установлен ли SSH
which ssh

# Проверяем статус службы
sudo systemctl status ssh

# ЕСЛИ НЕ УСТАНОВЛЕН (команда which ssh ничего не показала):
sudo apt install openssh-server -y

# ЕСЛИ УСТАНОВЛЕН, НО ВЫКЛЮЧЕН (статус inactive/dead):
sudo systemctl enable --now ssh

# Проверяем итоговый статус
sudo systemctl status ssh
# =============================================
# 5. НАСТРОЙКА ФАЙРВОЛА (UFW)
# =============================================
# Проверяем статус файрвола
sudo systemctl status ufw

# Включаем службу в автозагрузку
sudo systemctl enable ufw

# ВКЛЮЧАЕМ САМ ФАЙРВОЛ (обязательно!)
sudo ufw enable

# Проверяем статус (должен быть "активен")
sudo ufw statu
# =============================================
# 6. ОБНОВЛЕНИЕ ПАКЕТОВ
# =============================================
sudo apt update

Пункт 5:
«Выполнить резервное копирование установленной ОС и создать установочный образ системы»:
# =============================================
# 1. СОЗДАТЬ ПАПКУ ДЛЯ БЭКАПОВ
# =============================================
sudo mkdir -p /backup

# =============================================
# 2. СОЗДАТЬ ПОЛНУЮ РЕЗЕРВНУЮ КОПИЮ СИСТЕМЫ
# =============================================

sudo tar cvpzf /backup/backup_$(date +%Y%m%d).tgz \
    --exclude=/proc \
    --exclude=/lost+found \
    --exclude=/backup \
    --exclude=/mnt \
    --exclude=/sys \
    --exclude=/tmp \
    --exclude=/run \
    --exclude=/media \
    --exclude=/dev \
    /

# =============================================
# 3. ПРОВЕРИТЬ, ЧТО АРХИВ СОЗДАЛСЯ
# =============================================

ls -lh /backup/



Пункт 6:
«Создать точки восстановления системы»:

# =============================================
# 1. УСТАНОВКА TIMESHIFT
# =============================================
sudo apt update
sudo apt install timeshift -y

# =============================================
# 2. СОЗДАНИЕ ТОЧКИ ВОССТАНОВЛЕНИЯ
# =============================================
sudo timeshift --create --comments "Точка восстановления $(date)" --tags D

# =============================================
# 3. ПРОВЕРКА СПИСКА ТОЧЕК
# =============================================
sudo timeshift --list

# =============================================
# 4. (ДОПОЛНИТЕЛЬНО) ГДЕ ХРАНЯТСЯ СНИМКИ
# =============================================
ls -la /timeshift/

На случай если не хватило памяти: 
# =============================================
# 54.2 ТОЧКА ВОССТАНОВЛЕНИЯ (ручное создание)
# =============================================

# Создать папку для точек восстановления
sudo mkdir -p /restore_points

# Создать точку восстановления (копия /etc с датой)
sudo cp -r /etc /restore_points/restore_point_$(date +%Y%m%d_%H%M)

# Проверить, что точка создалась
ls -la /restore_points/

# Проверить размер
du -sh /restore_points/*




Пункт 7:
«Создать группы пользователей, настроить права доступа»:
# =============================================
# 1. СОЗДАТЬ ГРУППУ
# =============================================
sudo groupadd exam_group

# =============================================
# 2. СОЗДАТЬ ПОЛЬЗОВАТЕЛЯ И ДОБАВИТЬ В ГРУППУ
# =============================================
sudo useradd -m -G exam_group exam_user
sudo passwd exam_user

# =============================================
# 3. ПРОВЕРИТЬ, ЧТО ПОЛЬЗОВАТЕЛЬ СОЗДАН И В ГРУППЕ
# =============================================

id exam_user
getent group exam_group
groups exam_user

# =============================================
# 4. СОЗДАТЬ ПАПКУ И ФАЙЛ С ОГРАНИЧЕННЫМ ДОСТУПОМ
# =============================================

sudo mkdir /opt/exam_data
sudo chown $USER:$USER /opt/exam_data
sudo chmod 700 /opt/exam_data
echo "Секретные данные для экзамена" | sudo tee /opt/exam_data/secret.txt
sudo chmod 600 /opt/exam_data/secret.txt

# =============================================
# 5. ПРОВЕРИТЬ, ЧТО exam_user НЕ МОЖЕТ ПРОЧИТАТЬ ФАЙЛ
# =============================================

sudo su - exam_user -c "cat /opt/exam_data/secret.txt"

# =============================================
# 6. ПОКАЗАТЬ ПРАВА ДОСТУПА
# =============================================

ls -la /opt/exam_data/secret.txt

# =============================================
# 7. root МОЖЕТ ПРОЧИТАТЬ
# =============================================

sudo cat /opt/exam_data/secret.txt





Пункт 8:
«Настроить аутентификацию и авторизацию, журналов мониторинга»:

# =============================================
# 1. ПРОВЕРКА, ЧТО АУТЕНТИФИКАЦИЯ РАБОТАЕТ
# =============================================

# Проверить, что пользователь существует и имеет пароль
sudo cat /etc/shadow | grep exam_user

# =============================================
# 2. ПРОВЕРКА, ЧТО АВТОРИЗАЦИЯ РАБОТАЕТ
# =============================================

# Проверить, что пользователь может войти в систему
sudo su - exam_user -c "whoami && pwd"
# Должно показать: exam_user и /home/exam_user

# =============================================
# 3. ПРОВЕРКА, ЧТО ЛОГИ МОНИТОРИНГА РАБОТАЮТ
# =============================================
# Посмотреть последние записи в системном журнале
sudo journalctl -n 10

# ИЛИ (если journalctl недоступен)
sudo tail -n 10 /var/log/syslog
